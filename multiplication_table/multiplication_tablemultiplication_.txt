def multiplication_table(n, limit=10):
    print(f"Tabla de multiplicar del {n}:")
    for i in range(1, limit + 1):
        print(f"{n} x {i} = {n * i}")

def factorial_calculator(n):
    result = 1
    while n > 0:
        result *= n
        n -= 1
    return result

if __name__ == "__main__":
    num = int(input("Ingresa un número para la tabla de multiplicar: "))
    multiplication_table(num)

    fact_num = int(input("Ingresa un número para calcular su factorial: "))
    print(f"Factorial de {fact_num}: {factorial_calculator(fact_num)}")
